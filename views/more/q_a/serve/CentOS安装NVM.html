    • CentOS安装NVM
      • 下载安装
        • 注意： 如果安装了 zsh 则需要更改对应zsh的配置文件，而不是 ~/.bashrc
      • nvm 安装 node

# CentOS安装NVM

# 下载安装


# 安装 git
yum install git

# 下载 nvm
curl -o- https://raw.githubusercontent.com/nvm-sh/nvm/v0.35.2/install.sh | bash

# 安装完成后 copy 下面备用
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # This loads nvm
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"  # This loads nvm bash_completion

# 安装完成后在~/.bashrc等配置文件中会出现一下配置(如果没有请自己新增，并copy进去)
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # This loads nvm
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"  # This loads nvm bash_completion

# 运行配置文件
source ~/.bashrc

# ***** 配置环境变量 避免重启服务后 nvm 失效 *****

# 添加 .bash_profile
touch .bash_profile

# 编辑 .bash_profile, 将上面内容copy到文件中
vim .bash_profile

# 运行配置文件
source ~/.bash_profile


## 验证 nvm 
nvm -v

# 注意： 如果安装了 zsh 则需要更改对应zsh的配置文件，而不是 ~/.bashrc

# 修改 
vim  ~/.zshrc 

# 将一下copy到文件
export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && \. "$NVM_DIR/nvm.sh"  # This loads nvm
[ -s "$NVM_DIR/bash_completion" ] && \. "$NVM_DIR/bash_completion"  # This loads nvm bash_completion

# 运行配置
source ~/.zshrc

# nvm 安装 node

# 查看安装的node
nvm list

# 安装制定版本
nvm install 12.14.1

# 使用某个版本node
nvm use 12.14.1

# 将node设置为默认版本
nvm alias default 12.14.1
